##Usage

dissorted (arr, dir="a" , "d")

##Arguments

arr     Refers to the array that is being tested

"a"     Tests for ascending order 

"d"     Tests of descending order 

##Exampe

randomArray <- c(2, 3, 5, 6, 8, 9, 10)
dissorted (randomArray, dir="a")
